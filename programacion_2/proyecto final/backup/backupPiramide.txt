
    private double getBaseIngre(){
        boolean  bandera = true;

        textoIngresado = campoBaseP.getText();

        while(bandera){

            try{

                base = Integer.parseInt(textoIngresado);
                bandera = false;

            }catch(NumberFormatException nfe){

                JOptionPane.showMessageDialog(null, "Ingrese un número correcto, intente nuevamente", "Error", JOptionPane.ERROR_MESSAGE);

            }catch (Exception e){

                JOptionPane.showMessageDialog(null, "Error inesperado, intenta nuevamente");

            }

        }
        return base;
    }
